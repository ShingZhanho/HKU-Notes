\question
Let \( D \) be the set of complex numbers satisfying  
\[ \text{Im}(z) \leq \text{Re}(z) \quad \text{and} \quad |\text{Re}(z)| \leq 1. \]  
\begin{parts}
    \part For \( z = x + yi \), define \( D \) by a Cartesian equation in set notation.
    \begin{solution}
        $\text{Re}(z) = x$ and $\text{Im}(z) = y$, $x,y\in\mathbb{R}$. Thus,
        \[
            D = \boxed{\{(x,y)\mid y\leq x,\quad |x|\leq 1,\quad x,y\in\mathbb{R}\}}
        \]
    \end{solution}
    
    \part Sketch the set \( D \) in the complex plane.
    \begin{solution}
        By $y\leq x$, the region is the area below the line $y=x$ with boundary included.
        By $|x|\leq 1$, the region is bounded by the vertical lines $x=1$ and $x=-1$ with boundaries included.

    \end{solution}
\end{parts}
