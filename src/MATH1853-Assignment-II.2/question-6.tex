\question
Let $X$ be a discrete random variable with the probability distribution $p(x)$ given below.
\begin{center}
    \begin{tabular}{c|ccccc}
        $x$ & -2 & 0 & 1 & 3 & 4 \\
        \hline
        $p(x)$ & 0.1 & 0.3 & 0.2 & 0.25 & 0.15
    \end{tabular}
\end{center}
Compute $P(X\text{ is even})$ and $\mathbb{E}(X^2)$.

\begin{solution}
    We verify: $\sum_x p(x)=0.1+0.3+0.2+0.25+0.15=1$ and $p(x)\in[0,1]\forall x$, so $p(x)$ is a valid probability distribution.

    \begin{align*}
        P(X\text{ is even}) &= p(-2) + p(0) + p(4) \\
        &= 0.1 + 0.3 + 0.15 \\
        &= \boxed{0.55}. 
    \end{align*}

    \begin{align*}
        \mathbb{E}(X^2) &= \sum_x x^2 p(x) \\
        &= (-2)^2 \cdot 0.1 + 0^2 \cdot 0.3 + 1^2 \cdot 0.2 + 3^2 \cdot 0.25 + 4^2 \cdot 0.15 \\
        &= 4 \cdot 0.1 + 0 + 1 \cdot 0.2 + 9 \cdot 0.25 + 16 \cdot 0.15 \\
        &= 0.4 + 0 + 0.2 + 2.25 + 2.4 \\
        &= \boxed{5.25}.
    \end{align*}
\end{solution}
