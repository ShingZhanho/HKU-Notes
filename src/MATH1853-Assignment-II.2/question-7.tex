\question 
A random variable $X$ has the probability density function:
\[
    f(x) = \begin{cases}
        \frac{1}{36}x(6-x), &\text{for } 0 < x < 6, \\
        0, &\text{otherwise}.
    \end{cases}
\]
\begin{parts}
    \part Compute $P(1\leq X \leq 2)$.
    \begin{solution}
        We verify:
        \[
            \int_{-\infty}^{\infty} f(x) \,dx = \int_0^6 \frac{1}{36}x(6-x) \,dx = \frac{1}{36} \left[ 3x^2 - \frac{1}{3}x^3 \right]_0^6 = \frac{1}{36}(108 - 72) = 1.
        \]
        and $f(x)\geq 0$ for all $x$. Thus, $f(x)$ is a valid probability density function.

        Then,
        \begin{equation*}
            P(1\leq X \leq 2) = \int_1^2 f(x) \,dx
            = \int_1^2 \frac{1}{36}x(6-x) \,dx 
            = \frac{1}{36} \left[ 3x^2 - \frac{1}{3}x^3 \right]_1^2 
            = \boxed{\frac{5}{27}}.
        \end{equation*}
    \end{solution}

    \part Compute $\text{Var}(3X + 2)$.
    \begin{solution}
        \[
        \mathbb{E}(X) = \int_0^6 x f(x) \,dx
        = \int_0^6 x \cdot \frac{1}{36} x(6-x) \,dx
        = \frac{1}{36} \left[ 2x^3 - \frac{1}{4}x^4 \right]_0^6
        = \frac{1}{36}(432 - 324) = 3.
        \]
        \[
        \mathbb{E}(X^2) = \int_0^6 x^2 f(x) \,dx
        = \int_0^6 x^2 \cdot \frac{1}{36} x(6-x) \,dx
        = \frac{1}{36} \left[ \frac{3}{2}x^4 - \frac{1}{5}x^5 \right]_0^6
        = \frac{54}{5}.
        \]
        \[
        \text{Var}(X) = \mathbb{E}(X^2) - (\mathbb{E}(X))^2 = \frac{54}{5} - 9 = \frac{9}{5}.
        \]

        Thus, \[
        \text{Var}(3X + 2) = 9 \text{Var}(X) = 9 \cdot \frac{9}{5} = \boxed{\frac{81}{5}}.
        \]
    \end{solution}
\end{parts}
