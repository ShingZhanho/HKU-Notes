% augmented matrix environment
% #1: number of columns after the vertical line (default: 1)
% #2: number of columns before the vertical line
% #3: brace before the matrix (e.g. \left[ )
% #4: brace after the matrix (e.g. \right] )
\newenvironment{augmatrix}[4][1]{%
    \def\endbrace{#4}
    #3 \begin{array}{@{}*{#2}{c}|*{#1}{c}@{}}
}{%
    \end{array} \endbrace
}

\newenvironment{am41}{%
    \begin{augmatrix}{4}{\left[}{\right]}
}{%
    \end{augmatrix}
}
