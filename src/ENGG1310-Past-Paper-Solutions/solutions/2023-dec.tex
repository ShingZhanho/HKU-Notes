Number the gates as shown below:
\figHere{Complicated-NAND-Function.pdf}{width=0.47\textwidth}
Then, consider gates 1, 2, and 4, the subcircuit is expressed as:
\begin{align*}
    &\phantom{=}\overline{\overline{P\cdot Q}\cdot\overline{Q\cdot R}}\\
    &=P\cdot Q + Q\cdot R\\
    &=Q\cdot(P+R)
\end{align*}
Then, applying gate 5 on the same input, we obtain the negation 
\begin{align*}
    \overline{Q\cdot(P+R)} &= \overline{Q}+\overline{P+R} \\
    &= \overline{Q}+(\overline{P}\cdot\overline{R})
\end{align*}
Gate 3 gives $\overline{P\cdot R}=\overline{P}+\overline{R}$.
Finally, applying gate 6, we have over the two inputs, we have:
\begin{align*}
    &\phantom{=}\overline{\overline{Q}+(\overline{P}\cdot\overline{R}) \cdot (\overline{P}+\overline{R})}\\
    &=\overline{\overline{Q}+(\overline{P}\cdot\overline{R})} + \overline{\overline{P}+\overline{R}}\\
    &=(Q\cdot\overline{\overline{P}\cdot\overline{R}}) + (P\cdot R)\\
    &=(Q\cdot(P+R)) + (P\cdot R)\\
    &=(PQ+QR)+(PR)\\
    &=PQ+PR+QR
\end{align*}
Then, we draw the truth table:
\begin{center}

\begin{qsolution}{2023}{Dec}{Q24}{\tgEE \tgEEPowerSystems \tgEERMS}
    \mcqAns{A}

    From the graph, one can define
    \[
        v(t)=\begin{cases}
            \frac{6t}{T}, & t\in[0, 0.5T) \\
            6-\frac{6t}{T}, & t\in[0.5T, T)
        \end{cases}, \quad v(t+kT)=v(t), k\in\mathbb{Z}^+,
    \]
    Then, we can calculate the RMS voltage value by:
    \begin{align*}
        V_{\mathrm{RMS}} &= \sqrt{\frac{1}{T}\int_0^{T} v^2(t) \,\mathrm{d}t}\\
        &= \sqrt{\frac{1}{T}\left[\int_0^{T/2}\left(\frac{6t}{T}\right)^2\mathrm{d}t
            + \int_{T/2}^{T}\left(6-\frac{6t}{T}\right)^2\mathrm{d}t\right]}\\
        &= \sqrt{\frac{2}{T}\int_0^{T/2}\left(\frac{6t}{T}\right)^2\mathrm{d}t}\\
        &= \sqrt{\frac{72}{T^3}\int_0^{T/2} t^2\,\mathrm{d}t}\\
        &= \sqrt{\frac{72}{T^3}\cdot\frac{(T/2)^3}{3}}
        = \boxed{\sqrt{3}\,\mathrm{V}} \approx 1.732\,\mathrm{V}
    \end{align*}
\end{qsolution}

\begin{qsolution}{2023}{Dec}{LQ5}{\tgEE \tgEETransformer}
    The AC voltage source $V_H$ has a peak voltage at $t=8T$ of $900\sqrt 2$ V.
    Through the step-up transformer, the voltage on the secondary side $V_L$ is $\di V_L=\frac{1}{3}V_H=300\sqrt 2$ V.

    The equivalent resistance $R_{2||3}$ for the parallel connection $R_2$ and $R_3$ is $\di\left(\frac{1}{15}+\frac{1}{30}\right)^{-1}=10\,\Omega$.
    Then, the equivalent resistance $R_{eq}$ for the whole secondary side is $R_1+R_{2||3}=10+10=20\,\Omega$.

    $\di V_L=I_L\cdot R_{eq} \Rightarrow I_L=\frac{300\sqrt 2}{20}=15\sqrt 2$ A.

    Let $V_i$ and $I_i$ be the voltage and current flowing through the resistor $R_i$ ($i=1,2,3$).
    By KVL, we have $V_L=V_1+V_{2||3}$.
    Also, $V_1=I_L\cdot R_1=150\sqrt 2$ V.
    Therefore, $V_2=V_3=V_{2||3}=V_L-V_1=150\sqrt 2$ V.
    
    By Ohm's law, we have $\di I_2=\frac{V_2}{R_2}=10\sqrt 2$ A.

    Therefore, the peak current flowing through $R_2$ is $\boxed{10\sqrt 2\,\mathrm{A}} \approx 14.14\,\mathrm{A}$.
\end{qsolution}
